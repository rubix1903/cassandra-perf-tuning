STEP 3 — CQL Schema with Compaction Strategy

schema.cql 
-- Create keyspace with NetworkTopologyStrategy
CREATE KEYSPACE IF NOT EXISTS perf_test
  WITH REPLICATION = {
    'class': 'NetworkTopologyStrategy',
    'datacenter1': 3
  }
  AND DURABLE_WRITES = true;

USE perf_test;

-- Sensor events table (write-heavy → STCS)
CREATE TABLE IF NOT EXISTS sensor_events (
  device_id   UUID,
  event_time  TIMESTAMP,
  sensor_type TEXT,
  value       DOUBLE,
  metadata    MAP<TEXT, TEXT>,
  PRIMARY KEY (device_id, event_time)
) WITH CLUSTERING ORDER BY (event_time DESC)
  AND compaction = {
    'class': 'SizeTieredCompactionStrategy',
    'min_threshold': 4,
    'max_threshold': 32
  }
  AND compression = {
    'sstable_compression': 'LZ4Compressor',
    'chunk_length_in_kb': 64
  }
  AND caching = {
    'keys': 'ALL',
    'rows_per_partition': '100'
  };

-- User profile table (read-heavy → LCS + row cache)
CREATE TABLE IF NOT EXISTS user_profiles (
  user_id     UUID PRIMARY KEY,
  username    TEXT,
  email       TEXT,
  region      TEXT,
  created_at  TIMESTAMP
) WITH compaction = {
    'class': 'LeveledCompactionStrategy',
    'sstable_size_in_mb': 160
  }
  AND caching = {
    'keys': 'ALL',
    'rows_per_partition': 'NONE'   -- cache entire partition
  };
// 05 — STRESS TEST CODE

Python Benchmark Script

stress_test.py 
import uuid, time, statistics, threading, csv
from datetime import datetime, timezone
from cassandra.cluster import Cluster
from cassandra.policies import DCAwareRoundRobinPolicy, RetryPolicy
from cassandra import ConsistencyLevel
from cassandra.query import SimpleStatement, BatchStatement
from faker import Faker
from tqdm import tqdm
import pandas as pd
import matplotlib.pyplot as plt

# ─── CONFIG ─────────────────────────────────────────────────────
CONTACT_POINTS = ["127.0.0.1"]  # or EC2 IPs for AWS deploy
KEYSPACE       = "perf_test"
WRITE_OPS      = 50_000
READ_OPS       = 20_000
THREADS        = 16
BATCH_SIZE     = 50

fake = Faker()

class CassandraStressTest:
    def __init__(self):
        self.cluster = Cluster(
            contact_points=CONTACT_POINTS,
            load_balancing_policy=DCAwareRoundRobinPolicy(local_dc="datacenter1"),
            default_retry_policy=RetryPolicy(),
            protocol_version=5,
            connect_timeout=30
        )
        self.session = self.cluster.connect(KEYSPACE)
        self.session.default_consistency_level = ConsistencyLevel.QUORUM
        self.write_latencies = []
        self.read_latencies  = []
        self.errors = 0
        self._lock = threading.Lock()

        # Prepare statements (compiled once, reused)
        self.insert_stmt = self.session.prepare("""
            INSERT INTO sensor_events
              (device_id, event_time, sensor_type, value, metadata)
            VALUES (?, ?, ?, ?, ?)
            USING TTL 86400
        """)
        self.select_stmt = self.session.prepare("""
            SELECT * FROM sensor_events
            WHERE device_id = ?
            LIMIT 10
        """)

    def _write_batch(self, count):
        for _ in range(count):
            batch = BatchStatement()
            for _ in range(BATCH_SIZE):
                batch.add(self.insert_stmt, (
                    uuid.uuid4(),
                    datetime.now(timezone.utc),
                    fake.random_element(["temp", "humidity", "pressure"]),
                    fake.pyfloat(min_value=0, max_value=100),
                    {"location": fake.city(), "unit": fake.currency_code()}
                ))
            t0 = time.perf_counter()
            try:
                self.session.execute(batch)
                lat = (time.perf_counter() - t0) * 1000
                with self._lock:
                    self.write_latencies.append(lat)
            except Exception as e:
                with self._lock:
                    self.errors += 1

    def _read_batch(self, device_ids, count):
        for did in device_ids[:count]:
            t0 = time.perf_counter()
            try:
                rows = self.session.execute(self.select_stmt, (did,))
                _ = list(rows)
                lat = (time.perf_counter() - t0) * 1000
                with self._lock:
                    self.read_latencies.append(lat)
            except Exception:
                with self._lock:
                    self.errors += 1

    def run_write_stress(self):
        print(f"▶  Write stress: {WRITE_OPS * BATCH_SIZE:,} total rows via {THREADS} threads")
        t_start = time.time()
        threads = []
        per_thread = WRITE_OPS // THREADS
        for _ in range(THREADS):
            t = threading.Thread(target=self._write_batch, args=(per_thread,))
            threads.append(t); t.start()
        for t in tqdm(threads, desc="Writing"): t.join()
        elapsed = time.time() - t_start
        self._report("WRITE", self.write_latencies, elapsed)

    def run_read_stress(self, device_ids):
        print(f"▶  Read stress: {READ_OPS:,} lookups via {THREADS} threads")
        t_start = time.time()
        threads = []
        chunk = len(device_ids) // THREADS
        for i in range(THREADS):
            ids_chunk = device_ids[i*chunk:(i+1)*chunk]
            t = threading.Thread(target=self._read_batch,
                                 args=(ids_chunk, READ_OPS // THREADS))
            threads.append(t); t.start()
        for t in tqdm(threads, desc="Reading"): t.join()
        elapsed = time.time() - t_start
        self._report("READ", self.read_latencies, elapsed)

    def _report(self, op, latencies, elapsed):
        if not latencies: return
        lats = sorted(latencies)
        print(f"\n{'─'*50}")
        print(f"  {op} RESULTS")
        print(f"  Total ops   : {len(latencies):,}")
        print(f"  Throughput  : {len(latencies)/elapsed:,.0f} ops/sec")
        print(f"  Mean latency: {statistics.mean(lats):.2f} ms")
        print(f"  p50 latency : {lats[int(len(lats)*0.50)]:.2f} ms")
        print(f"  p95 latency : {lats[int(len(lats)*0.95)]:.2f} ms")
        print(f"  p99 latency : {lats[int(len(lats)*0.99)]:.2f} ms")
        print(f"  Errors      : {self.errors}")
        print(f"{'─'*50}\n")

    def plot_latencies(self):
        fig, axes = plt.subplots(1, 2, figsize=(14, 5))
        fig.patch.set_facecolor('#0a0c10')
        for ax, data, title in zip(
            axes,
            [self.write_latencies, self.read_latencies],
            ["Write Latency (ms)", "Read Latency (ms)"]
        ):
            ax.set_facecolor('#0f1318')
            ax.hist(data, bins=60, color='#00d4ff', alpha=0.75, edgecolor='none')
            ax.set_title(title, color='#e2ecf4', pad=12)
            ax.tick_params(colors='#5a7a96')
            for spine in ax.spines.values(): spine.set_color('#1e2a38')
        plt.tight_layout()
        plt.savefig("latency_distribution.png", dpi=150, bbox_inches='tight')
        print("  Saved: latency_distribution.png")

    def close(self): self.cluster.shutdown()


if __name__ == "__main__":
    test = CassandraStressTest()
    sample_ids = [uuid.uuid4() for _ in range(1000)]
    test.run_write_stress()
    test.run_read_stress(sample_ids)
    test.plot_latencies()
    test.close()
// 06 — PERFORMANCE TUNING

Optimization Playbook

Compaction Strategy Selection

STCS (SizeTieredCompactionStrategy) — Default. Best for write-heavy workloads. Merges similarly-sized SSTables. Lower write amplification, higher read amplification.

LCS (LeveledCompactionStrategy) — Best for read-heavy workloads. Organizes SSTables into levels. Predictable read performance, ~10× higher write I/O.

TWCS (TimeWindowCompactionStrategy) — For time-series. Groups SSTables by time window. Perfect for sensor data with TTL.

Row Cache Tuning

Row cache stores entire deserialized rows in memory — huge wins for hot partitions. Enable only on read-heavy tables with small partition sizes.

Set in cassandra.yaml: row_cache_size_in_mb: 512
Per-table: caching = {'keys':'ALL', 'rows_per_partition':'NONE'}

Monitor with: nodetool info | grep -i cache

AWS EBS gp3 Volume Configuration

ebs_provision.sh — provision gp3 volumes via AWS CLI 
#!/bin/bash
# Create a 500GB gp3 EBS volume optimized for Cassandra
# gp3 key advantage: IOPS and throughput independent of size

ZONE="us-east-1a"

# Create volume: 500GB gp3, 6000 IOPS, 250MB/s throughput
VOLUME_ID=$(aws ec2 create-volume \
  --availability-zone $ZONE \
  --volume-type gp3 \
  --size 500 \
  --iops 6000 \
  --throughput 250 \
  --tag-specifications \
    'ResourceType=volume,Tags=[{Key=Name,Value=cassandra-node1}]' \
  --query 'VolumeId' --output text)

echo "Created: $VOLUME_ID"

# Attach to EC2 instance
aws ec2 attach-volume \
  --volume-id $VOLUME_ID \
  --instance-id i-XXXXXXXXXXXXXXXXX \
  --device /dev/xvdf

# Format and mount (on the EC2 instance)
# mkfs.ext4 -E nodiscard /dev/xvdf
# mount -o noatime,nodiscard /dev/xvdf /var/lib/cassandra
# echo "/dev/xvdf /var/lib/cassandra ext4 noatime,nodiscard 0 2" >> /etc/fstab

# gp3 tuning note: nodiscard prevents TRIM from slowing writes
# noatime prevents access-time updates on every read = major I/O win

Key Wins Explained

Row cache eliminated nearly all disk reads for hot user_profiles — rows served from RAM at nanosecond speed.
LCS compaction reduced SSTable count from 300+ to ~18, meaning far fewer files to merge on reads.
noatime + nodiscard mount flags cut EBS write amplification significantly.
gp3 at 6000 IOPS sustained throughput without the credit-based throttling of gp2.
